### AI 算法逻辑

**评分系统**：

1. **血压评分** (100 分制)
```javascript
function calculateBPScore(systolic, diastolic) {
    let score = 100;
    const normalSystolic = 120;
    const normalDiastolic = 80;
    
    // 与正常值的偏差越大，扣分越多
    score -= Math.abs(systolic - normalSystolic) * 1.5;
    score -= Math.abs(diastolic - normalDiastolic) * 1.5;
    
    return Math.max(0, Math.min(100, Math.round(score)));
}
```

2. **血糖评分** (100 分制)
```javascript
function calculateSugarScore(bloodSugar) {
    let score = 100;
    const normalSugar = 5.5;
    
    // 血糖偏离正常值扣分
    score -= Math.abs(bloodSugar - normalSugar) * 10;
    
    // 血糖过高额外扣分
    if (bloodSugar >= 7.0) score -= 20;
    
    return Math.max(0, Math.min(100, Math.round(score)));
}
```

3. **生活方式评分** (100 分制)
```javascript
function calculateLifestyleScore(sleep, sport, weight) {
    let score = 100;
    
    // 睡眠评分
    if (sleep < 6 || sleep > 9) score -= 15;
    else if (sleep < 7 || sleep > 8) score -= 5;
    
    // 运动评分
    if (sport < 30) score -= 20;
    else if (sport < 60) score -= 10;
    
    // 体重评分
    if (weight > 140 || weight < 90) score -= 15;
    else if (weight > 120 || weight < 100) score -= 5;
    
    return Math.max(0, Math.min(100, Math.round(score)));
}
```

function generateSuggestions(avg, latest, trend) {
    const suggestions = [];
    
    // 血压建议
    if (avg.systolic > 130 || avg.diastolic > 85) {
        suggestions.push({
            title: '🫀 血压管理',
            content: '您的血压偏高，建议：低盐饮食...'
        });
    }
    
    // 血糖建议
    if (avg.bloodSugar > 6.0) {
        suggestions.push({
            title: '🍬 血糖控制',
            content: '您的血糖偏高，建议：控制碳水...'
        });
    }
    
    // 睡眠建议
    if (avg.sleep < 7) {
        suggestions.push({
            title: '😴 睡眠改善',
            content: '您的睡眠不足，建议：保持规律...'
        });
    }
    
    // 运动建议
    if (avg.sport < 30) {
        suggestions.push({
            title: '🏃 运动建议',
            content: '您的运动量不足，建议：每天...'
        });
    }
    
    // 如果没有问题，显示鼓励
    if (suggestions.length === 0) {
        suggestions.push({
            title: '🎉 健康状况良好',
            content: '您的各项指标都在正常范围内...'
        });
    }
    
    return suggestions;
}
